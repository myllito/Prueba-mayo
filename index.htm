            // --- Update Chart Functions ---

            function updateDetailChart(chart, data, unit, max) {

                 chart.data.labels = data.map(d => d.name.length > 25 ? d.name.substring(0, 25) + '...' : d.name);

                 chart.data.datasets[0].data = data.map(d => d.value);

                 chart.data.datasets[0].backgroundColor = data.map(d => d.value < 80 ? '#38bdf8' : (d.value < 95 ? '#f59e0b' : '#ef4444'));

                 chart.options.scales.x.max = max;

                 chart.options.plugins.tooltip.callbacks.label = function(context) {

                     let label = context.dataset.label || '';

                     if (label) {

                         label += ': ';

                     }

                     if (context.parsed.x !== null) {

                         label += `${context.parsed.x.toFixed(2)} ${unit}`;

                     }

                     return label;

                 };

                 chart.update();

            }

            const tabContents = document.querySelectorAll('.tab-content');

                    navTabs.forEach(t => t.classList.remove('active-tab'));

                    tab.classList.add('active-tab');

 

                    const target = tab.getAttribute('data-target');

                    tabContents.forEach(content => {

                        if (content.id === target) {

                            content.classList.add('active');

                        } else {

                            content.classList.remove('active');

                        }

                    });

                });

            });

                        b.classList.add('bg-white', 'text-slate-700');

                    });

                    btn.classList.remove('bg-white', 'text-slate-700');

                    const type = btn.getAttribute('data-type');
